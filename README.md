# HttpInfo


Usage: httpinfo.exe -p 80 -b 192.168.1.1 -e 192.168.1.254 -t 10 -o 5


  -b string
  
        Begin IP
        
  -e string
  
        End IP
        
  -i string
  
        Input from file, beginip and endip or file must be have one
        
  -o int
  
        Timeout settings (default 5)
        
  -p int
  
        Port (default 80)
        
  -t int
  
        Thread worker to scan (default 10)
